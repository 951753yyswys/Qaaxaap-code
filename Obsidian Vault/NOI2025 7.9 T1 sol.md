		c&\equiv 2d+e\pmod{n}\\
		d&\equiv 2e\pmod{n}\\

		3d&\equiv 2c+d\pmod{n}\\
		0&\equiv 3d\pmod{n}\\
		0&\equiv 2c+d\pmod{n}\\
	\end{aligned}
\end{align}
\right.
$$

$\Rightarrow e\equiv 7e\equiv 0\pmod{n}$
